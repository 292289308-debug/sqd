🚀 快速开始

# 克隆仓库
git clone https://github.com/292289308-debug/sqd.git
cd sqd/backend

# 安装依赖
pip install -r requirements.txt

# 配置 Tushare token (可选)
cp .env.example .env
# 编辑 .env 填入 TUSHARE_TOKEN=***

# 拉取真实 K 线数据
cd ..
python scripts/fetch_top100.py
python scripts/fetch_skip_basic.py

# 启动后端
cd backend
uvicorn app.main:app --host 0.0.0.0 --port 8000 --reload

# 浏览器访问 http://localhost:8000
